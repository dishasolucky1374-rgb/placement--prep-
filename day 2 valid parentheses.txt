#include<stdbool.h>
#include<string.h>


bool isValid(char* s) {
 char stack[100];
 int top = -1;
 int i;
 for(i=0;i<strlen(s);i++){
    if(s[i]=='('||s[i]=='['||s[i]=='{'){
          stack[++top]= s[i];
    }else{
        if(top == -1) return false;

        if((s[i]==')' && stack[top]=='(') ||
               (s[i]==']' && stack[top]=='[') ||
               (s[i]=='}' && stack[top]=='{')) {
                top--;
        }else{
                return false;
               }
    }
    }
    return top == -1;
}
